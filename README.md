# modulo-1-devinhouse

<!-- criar o projeto -->
npm init vue@latest

<!-- instalar vuetify e mdifonts -->
npm install vuetify@^3.3.6
npm install @mdi/font -D  

<!-- instalar vuex, variaveis globais -->
npm install vuex@next --save

<!-- instalar axios -->
npm install axios

<!-- instalar node servidor e frontend -->
npm install       

<!-- iniciar servidor -->
npm run server 

<!-- iniciar frontend -->
npm run dev
